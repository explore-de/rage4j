import CodeBlock from '@theme/CodeBlock';
import {getGitHubStats} from '@site/src/api/GitHubApi';

export const FetchedDependency = () => {
  const {stats, loading} = getGitHubStats('explore-de', 'rage4j');

  const getMavenDependency = (version) => `<dependency>
    <groupId>dev.rage4j</groupId>
    <artifactId>rage4j-assert</artifactId>
    <version>${version}</version>
</dependency>`;

  return (
    <CodeBlock language="xml">
      {getMavenDependency(loading ? '1.0.2' : stats.version)}
    </CodeBlock>
  );
};

## Installation: Add Rage4J-Assert to Your Project

To start using Rage4J, add the following Maven dependency to your `pom.xml` file:

<FetchedDependency />

Once added, Maven will download and include the library in your project automatically.
